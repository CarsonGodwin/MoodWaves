int streakOfDates(List<DateTime> dates) {
  final datesSet = dates.toSet();

  int currentStreak = 1;

  for (final date in datesSet) {
    if (datesSet.contains(date.subtract(const Duration(days: 1)))) {
      currentStreak++;
    } else {
      return currentStreak;
    }
  }

  return currentStreak;
}
